# 発問の木 データベース：スキーマ と サンプル

122授業ぶんを、下のJSONスキーマで `src/data/dialogue/questionTrees.json` に格納する。
1授業＝1オブジェクト。中心発問（背骨）＋補助発問（階段）＋タグ（学習内容・想定反応・誤概念）。

## スキーマ

```jsonc
{
  "id": "c1-1",                 // questionBank と同じID
  "chapterId": "c1",
  "daisetsu": "1 整数の性質",    // 大節
  "shosetsu": "素数",            // 小節＝小学習内容（タグの中心）
  "jikan": "2時間",
  "page": "014",
  "hatsumon": "（原文の学習課題）",
  "nerai": "（原文のねらい）",
  "tags": {
    "unit": "整数の性質",         // 単元
    "subunit": "素数",            // 小学習内容
    "cognitive": "知識・技能",     // 知識・技能 / 思考・判断・表現
    "ruleType": ["仲間分け→定義化"], // 主に使う発問の型（①目的願望/②類推/③揺さぶり/④アハ など）
    "skills": ["約数で分類する", "素数の判定"]  // つまずき診断用スキルタグ
  },
  "central": [                    // 中心発問（スムーズな子はここだけ通る）
    {
      "no": 1,
      "prompt": "（発問の言葉）",
      "intent": "（この発問のねらい）",
      "answerType": "free",       // free（自由記述）/ handwrite（手書き）/ choice（選択）
      "board": "（黒板に残すもの）",
      "skipIf": "（飛ばし条件：この答えが出たら次の山場へ）",
      "expected": ["（予想される反応）", "..."],
      "misconceptions": [          // 想定誤答 × 背後の誤概念
        { "wrong": "（誤答）", "concept": "（誤概念）" }
      ],
      "aids": [                    // 補助発問（躓いたら上から降ろす階段）
        { "step": 1, "type": "既習比較/焦点化", "prompt": "..." },
        { "step": 2, "type": "具体化/視覚化", "prompt": "...", "choices": ["①..","②.."] },
        { "step": 3, "type": "二者択一/4択", "prompt": "...", "choices": ["①..","②.."] },
        { "step": "説明", "type": "説明", "prompt": "（定義・結論はここで普通に教える）" }
      ]
    }
  ]
}
```
